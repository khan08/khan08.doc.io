[docs]def launch(n, instance_type, keyname, key_filename, image_id=ec2Config.IMAGE_ID, sgs=ec2Config.SECURITY_GROUPS):
    """
    This function helps user to launch a dask cluster on aws.
    It will launch a t2.micro type instance as the manager node, and launch custom instances as worker nodes.

    :param n: The number of worker instances
    :param instance_type: The type of the worker instances
    :param keyname: The ssh key name of the manager instance
    :param key_filename: The .pem path. For example, /.ssh/mykey.pem
    :param image_id: The operating system of the worker instances
    :param sgs: The security group of the manager and worker instance
    :return: returns manager instance with definition as `boto3.EC2.instance <http://boto3.readthedocs.io/en/latest/reference/services/ec2.html#instance>`_
    """
    ec2 = boto3.resource('ec2')
    client = boto3.client("ec2")
    session = boto3.Session()
    credentials = session.get_credentials()
    user_data = ec2Config.USER_SCRIPT.format(
        n=n, instance_type=instance_type, access_key=credentials.access_key, secret_key=credentials.secret_key
    )
    instances = ec2.create_instances(
        MinCount=1,
        MaxCount=1,
        KeyName=keyname,
        SecurityGroups=sgs,
        UserData=user_data,
        ImageId=image_id,
        InstanceType=ec2Config.INSTANCE_TYPE,
    )
    print ("Creating Instance")
    print ("Wait till instance is running")
    waiter = client.get_waiter("instance_running")
    try:
        waiter.wait(InstanceIds=[instances[0].id])
    except WaiterError:
        raise Exception(
            "An unexpected error occurred when launching the requested nstances.")
    instances[0].load()

    print ("Wait till instance ssh ready")
    wait_ssh_ready(instances[0].public_ip_address)

    print ("Startup script logs:")
    command = 'tail -f /var/log/cloud-init-output.log'
    ssh(command,instances[0].public_ip_address,key_filename)
    return instances[0]



[docs]def ssh(command,host,key_filename):
    import paramiko
    ssh = paramiko.SSHClient()
    ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
    ssh.connect(hostname=host,
                username='ubuntu',
                port=22,
                key_filename=key_filename,
                compress=True,
                timeout=20,
                banner_timeout=20)
    client = ssh.get_transport()
    session = client.open_channel(kind='session')
    session.exec_command(command)
    while True:
        if session.recv_ready():
            print session.recv(4096)
        if session.recv_stderr_ready():
            print session.recv_stderr(4096)
        if session.exit_status_ready():
            break

    print 'exit status: ', session.recv_exit_status()
    session.close()
    client.close()
    ssh.close()
